Diferencias entre =, == y ===

Operadores de Asignación y Comparación


=
El operador de asignación = se utiliza para asignar un valor a una variable.
let a = 5; // Asigna el valor 5 a la variable a

==
El operador de comparación == compara dos valores para igualdad después de convertir ambos valores a un tipo común (coerción de tipo).
5 == "5"; // true, porque convierte "5" a número antes de comparar
0 == false; // true, porque convierte false a 0 antes de comparar

===
El operador de comparación estricta === compara tanto el valor como el tipo sin realizar coerción de tipo.
5 === "5"; // false, porque no son del mismo tipo
0 === false; // false, porque no son del mismo tipo
5 === 5; // true, porque son del mismo tipo y valor

Avanzando con Más Conceptos

Funciones Flecha


// Función tradicional
function sumar(a, b) {
  return a + b;
}

// Función flecha
const sumar = (a, b) => a + b;

console.log(sumar(2, 3)); // 5

Destructuración


// Destructuración de arreglos
let [x, y] = [1, 2];
console.log(x); // 1
console.log(y); // 2

// Destructuración de objetos
let { nombre, edad } = { nombre: "Ana", edad: 28 };
console.log(nombre); // Ana
console.log(edad); // 28

Plantillas de Cadenas de Texto


let nombre = "Juan";
let mensaje = `Hola, ${nombre}!`;
console.log(mensaje); // Hola, Juan!

Spread Operator


// En arreglos
let arr1 = [1, 2, 3];
let arr2 = [...arr1, 4, 5, 6];
console.log(arr2); // [1, 2, 3, 4, 5, 6]

// En objetos
let obj1 = { a: 1, b: 2 };
let obj2 = { ...obj1, c: 3 };
console.log(obj2); // { a: 1, b: 2, c: 3 }

Asincronía y async/await


// Función asincrónica
async function obtenerDatos() {
  try {
    let response = await fetch('https://api.example.com/datos');
    let data = await response.json();
    console.log(data);
  } catch (error) {
    console.error('Error:', error);
  }
}

obtenerDatos();
